\section{Bài 2}

\subsection{Đề bài}

Cho $x, y \in \mathbb{C}$, chứng minh:
\begin{itemize}
    \item[(a)] $x \overline{x} = \overline{x} x = |x|^2$.
    \item[(b)] $\overline{x^{-1}} = \left(\overline{x}\right)^{-1} (x \neq 0)$.
    \item[(c)] $|xy| = |x||y|$.
    \item[(d)] $|x + y| \leq |x| + |y|$.
\end{itemize}

\subsection{Lời giải}
\subsubsection{Phần (a)}
Giả sử $x = a + bi, (a, b \in \mathbb{R})$, ta có:
\[
\begin{cases}
    x \overline{x} = (a + bi)(a - bi)\\
    \overline{x} x = (a - bi)(a + bi) = (a + bi)(a - bi)
\end{cases} \Rightarrow x \overline{x} = \overline{x} x
\]
Lại có:
\[
x \overline{x} = (a + bi)(a - bi) = a^2 - (bi)^2 = a^2 + b^2
\]
mà $|x|^2 = \left(\sqrt{a^2 + b^2}\right)^2 = a^2 + b^2$
\\
$\Rightarrow x \overline{x} = |x|^2 \Leftrightarrow x \overline{x} = \overline{x} x = |x|^2 \Rightarrow$ dpcm.

\subsubsection{Phần (b)}
Giả sử $x = a + bi, (a, b \in \mathbb{R} \backslash \{0\})$, ta có:
\[
\overline{x^{-1}} = \overline{\frac{\overline{x}}{|x|^2}} = \frac{x}{|x|^2}
\]
\[
\left(\overline{x}\right)^{-1} = \frac{\overline{\overline{x}}}{|x|^2} = \frac{x}{|x|^2}
\]

$\Rightarrow \overline{x^{-1}} = \left(\overline{x}\right)^{-1} = \dfrac{x}{|x|^2} \Rightarrow$ dpcm.

\subsubsection{Phần (c)}
Giả sử $x = a + bi, y = c + di, (a, b, c, d \in \mathbb{R})$, ta có:
\[
\begin{cases}
    |x| = \sqrt{a^2 + b^2} \\
    |y| =  \sqrt{c^2 + d^2}\\
    xy = (a + bi)(c + di) = (ac - bd) + (ad + bc)i
\end{cases}
\]
\[
\Rightarrow
\begin{cases}
    |x||y| = (\sqrt{a^2 + b^2})(\sqrt{c^2 + d^2}) = \sqrt{(ac)^2 + (bc)^2 + (ad)^2 + (bd)^2} \\
    |xy| = \sqrt{(ac - bd)^2 + (ad + bc)^2} = \sqrt{(ac)^2 + (bc)^2 + (ad)^2 + (bd)^2}
\end{cases}
\]
$\Rightarrow |x||y| = |xy| = \sqrt{(ac)^2 + (bc)^2 + (ad)^2 + (bd)^2} \Rightarrow$ dpcm.

\subsubsection{Phần (d)}
Giả sử $x = a + bi, y = c + di, (a, b, c, d \in \mathbb{R})$, ta có:
\[
\begin{cases}
    |x| = \sqrt{a^2 + b^2} \\
    |y| = \sqrt{c^2 + d^2} \\
    |x + y| = \sqrt{(a + c)^2 + (b + d)^2}
\end{cases}
\]
Xét mệnh đề:
\begin{align*}
    |x + y| &\leq |x| + |y| \\
    \Leftrightarrow \sqrt{(a + c)^2 + (b + d)^2} &\leq \sqrt{a^2 + b^2} + \sqrt{c^2 + d^2} \\
    \Leftrightarrow (a + c)^2 + (b + d)^2 &\leq \left(\sqrt{a^2 + b^2} + \sqrt{c^2 + d^2}\right)^2 \\
    \Leftrightarrow a^2 + 2ac + c^2 + b^2 + 2bd + d^2 &\leq a^2 + b^2 + c^2 + d^2 + 2\sqrt{(a^2 + b^2)(c^2 + d^2)} \\
    \Leftrightarrow \sqrt{(a^2 + b^2)(c^2 + d^2)} &\geq ac + bd, \ \text{(luôn đúng $\forall a,b, c, d \in \mathbb{R}$)} 
\end{align*}
$\Rightarrow \text{mệnh đề được xét đúng} \Rightarrow |x + y| \leq |x| + |y| \Rightarrow$ dpcm.
