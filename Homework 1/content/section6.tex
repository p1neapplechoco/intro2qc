\section{Bài 6}

\subsection{Đề bài}

Cho $\ket{\phi} = \frac{1}{2}\ket{00} + \frac{i}{\sqrt{2}}\ket{10} + \frac{\sqrt{3} + i}{4}\ket{11}$
\begin{itemize}
    \item[(a)] Cho thấy $\ket{\phi}$ là vector đơn vị.
    \item[(b)] Tính $\operatorname{proj}_{\ket{+-}}{\ket{\phi}}$ và chuẩn hoá $\operatorname{proj}_{\ket{+-}}{\ket{\phi}}$.
    \item[(c)] Tính toạ độ của $\ket{\phi}$ theo cơ sở Bell.
\end{itemize}

\subsection{Lời giải}
\subsubsection{Phần (a)}
Ta có:
\begin{align*}
    \ket{\phi} &= \frac{1}{2}\ket{00} + \frac{i}{\sqrt{2}}\ket{10} + \frac{\sqrt{3} + i}{4}\ket{11} \\
    &= \frac{1}{2}\begin{bmatrix} 1 \\ 0 \\ 0 \\ 0 \end{bmatrix} + \frac{i}{\sqrt{2}}\begin{bmatrix} 0 \\ 0 \\ 1 \\ 0 \end{bmatrix} + \frac{\sqrt{3} + i}{4}\begin{bmatrix} 0 \\ 0 \\ 0 \\ 1 \end{bmatrix} \\
    \Rightarrow \ket{\phi} &= \begin{bmatrix} \frac{1}{2} \\ 0 \\ \frac{i}{\sqrt{2}} \\ \frac{\sqrt{3} + i}{4} \end{bmatrix}
\end{align*}
\[
\Rightarrow \|\ket{\phi}\| = \sqrt{\left|\frac{1}{2}\right|^2 + \left|\frac{i}{\sqrt{2}}\right|^2 + \left| \frac{\sqrt{3} + i}{4}\right|^2} = \sqrt{\frac{1}{4} + \frac{1}{2} + \frac{1}{4}} = 1 
\]
$\Rightarrow \ket{\phi} $ là vector đơn vị.

\subsubsection{Phần (b)}
Ta có:
\[
\ket{+-} = \frac{1}{2}\begin{bmatrix} 1 \\ -1 \\ 1 \\ -1 \end{bmatrix} \Rightarrow \bra{+-} = \frac{1}{2}\begin{bmatrix} 1 & -1 & 1 & -1 \end{bmatrix}.
\]
\begin{align*}
    \braket{+-}{\phi} &= \frac{1}{2}\begin{bmatrix} 1 & -1 & 1 & -1 \end{bmatrix} \begin{bmatrix} \frac{1}{2} \\ 0 \\ \frac{i}{\sqrt{2}} \\ \frac{\sqrt{3} + i}{4} \end{bmatrix} \\
    &= \frac{1}{2}\left(\frac{1}{2} + \frac{i}{\sqrt{2}} - \frac{\sqrt{3} + i}{4}\right) \\
    &= \frac{1}{2}\left(\frac{2 - \sqrt{3}}{4} + i\left(\frac{1}{\sqrt{2}} - \frac{1}{4}\right)\right) \\
    &= \frac{1}{2}\left(\frac{2 - \sqrt{3}}{4} + i\frac{4 - \sqrt{2}}{4\sqrt{2}}\right) \\
    &= \frac{2 - \sqrt{3}}{8} + i\frac{4 - \sqrt{2}}{8\sqrt{2}}
\end{align*}
\begin{align*}
    \Rightarrow \operatorname{proj}_{\ket{+-}}{\ket{\phi}} &= \frac{\braket{+-}{\phi}}{\|\ket{+-}\|^2}\ket{+-} \\
    &= \left(\frac{2 - \sqrt{3}}{8} + i\frac{4 - \sqrt{2}}{8\sqrt{2}}\right) \ket{+-} \\
\end{align*}
Chuẩn hoá $\operatorname{proj}_{\ket{+-}}{\ket{\phi}}$, ta được:
\[
e = \frac{\operatorname{proj}_{\ket{+-}}{\ket{\phi}}}{\|\operatorname{proj}_{\ket{+-}}{\ket{\phi}}\|} = \frac{(2 - \sqrt{3}) + i\frac{4 - \sqrt{2}}{\sqrt{2}}}{\sqrt{(2 - \sqrt{3})^2 + \frac{(4 - \sqrt{2})^2}{2}}} \ket{+-}
\]

\subsubsection{Phần (c)}
Cơ sở Bell là:
\[
B = \left\{\ket{\Phi^+} = \begin{bmatrix} \frac{1}{\sqrt{2}} \\ 0 \\ 0  \\ \frac{1}{\sqrt{2}} \end{bmatrix}, \ket{\Phi^-} = \begin{bmatrix} \frac{1}{\sqrt{2}} \\ 0 \\ 0  \\ -\frac{1}{\sqrt{2}} \end{bmatrix}, \ket{\Psi^+} = \begin{bmatrix} 0 \\ \frac{1}{\sqrt{2}} \\ \frac{1}{\sqrt{2}} \\ 0 \end{bmatrix}, \ket{\Psi^-} = \begin{bmatrix} 0 \\ \frac{1}{\sqrt{2}} \\ -\frac{1}{\sqrt{2}} \\ 0 \end{bmatrix}\right\}
\]
Ta có:
\begin{align*}
    \braket{\Phi^+}{\phi} &= \frac{1}{\sqrt{2}}\left(\frac{1}{2} + \frac{\sqrt{3} + i}{4}\right) = \frac{1}{\sqrt{2}} \cdot \frac{2 + \sqrt{3} + i}{4} = \frac{2 + \sqrt{3} + i}{4\sqrt{2}} \\
    \braket{\Phi^-}{\phi} &= \frac{1}{\sqrt{2}}\left(\frac{1}{2} - \frac{\sqrt{3} + i}{4}\right) = \frac{1}{\sqrt{2}} \cdot \frac{2 - \sqrt{3} - i}{4} = \frac{2 - \sqrt{3} - i}{4\sqrt{2}} \\
    \braket{\Psi^+}{\phi} &= \frac{1}{\sqrt{2}}\left(0 + \frac{i}{\sqrt{2}}\right) = \frac{i}{2} \\
    \braket{\Psi^-}{\phi} &= \frac{1}{\sqrt{2}}\left(0 - \frac{i}{\sqrt{2}}\right) = -\frac{i}{2}
\end{align*}
\begin{align*}
    \Rightarrow \left[\ket{\phi}\right]_B &= \left(\braket{\Phi^+}{\phi}, \braket{\Phi^-}{\phi}, \braket{\Psi^+}{\phi}, \braket{\Psi^-}{\phi}\right) \\
    &= \left(\frac{2 + \sqrt{3} + i}{4\sqrt{2}}, \frac{2 - \sqrt{3} - i}{4\sqrt{2}}, \frac{i}{2}, -\frac{i}{2}\right)
\end{align*}

$\Rightarrow$ toạ độ của $\ket{\phi}$ trong cơ sở Bell là $\begin{bmatrix}\frac{2 + \sqrt{3} + i}{4\sqrt{2}} \\ \frac{2 - \sqrt{3} - i}{4\sqrt{2}} \\ \frac{i}{2} \\ -\frac{i}{2} \end{bmatrix}$.
